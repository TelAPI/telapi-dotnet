using System;
using TelAPI;

namespace TelAPI.Example
{
    public class PlayAudioToMember
    {
        public static void Main(string[] args)
        {
            var client = new TelAPIRestClient("your-account-sid", "your-auth-token");

            try
            {
                var conference = client.GetConference("conference-sid");
                var result = client.PlayAudioToMember(conference.Sid, conference.Members[0], "http://audio-url.com");

                Console.WriteLine("Conference name : {0}", result.Name);
            }
            catch (TelAPIException ex)
            {
                Console.WriteLine("{0}", ex.Message); 
            }
        }
    }
}
